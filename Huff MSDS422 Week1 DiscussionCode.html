# create Survived dataframe with ticket class as factor (1st, 2nd, 3rd),
#    broken into male and female subfactors

# general counts
allMales = df3[df3.Gender==0].count()['Gender']
allSurvivedMales = df3[(df3.Gender==0) & (df3.Survived==1)].count()['Survived']
allFemales = df3[df3.Gender==1].count()['Gender']
allSurvivedFemales = df3[(df3.Gender==1) & (df3.Survived==1)].count()['Survived']
allPassengers = allMales + allFemales

# counts of Gender given Ticket Class
m1 = df3[(df3.Gender==0) & (df3.Pclass==1)].count()['Gender']
f1 = df3[(df3.Gender==1) & (df3.Pclass==1)].count()['Gender']
m2 = df3[(df3.Gender==0) & (df3.Pclass==2)].count()['Gender']
f2 = df3[(df3.Gender==1) & (df3.Pclass==2)].count()['Gender']
m3 = df3[(df3.Gender==0) & (df3.Pclass==3)].count()['Gender']
f3 = df3[(df3.Gender==1) & (df3.Pclass==3)].count()['Gender']

# counts of being in a certain class
first = df3[df3.Pclass==1].count()['Pclass']
second = df3[df3.Pclass==2].count()['Pclass']
third = df3[df3.Pclass==3].count()['Pclass']
allClasses = first + second + third
#------------------------
# probability of being in a certain class
p1st = 1.0 * first/allClasses
p2nd = 1.0 * second/allClasses
p3rd = 1.0 * third/allClasses

# probability of being a certain gender, given you are in a certain class
pM_1 = 1.0 * m1/first
pF_1 = 1.0 * f1/first
pM_2 = 1.0 * m2/second
pF_2 = 1.0 * f2/second
pM_3 = 1.0 * m3/third
pF_3 = 1.0 * f3/third

# probability of survival, given you are a certain gender (M/F)
pS_M = 1.0 * allSurvivedMales/allMales
pS_F = 1.0 * allSurvivedFemales/allFemales
#--------------
# probability of Survival, given you are a certain Gender in a certain Ticket Class

pS_M1 = pS_M * pM_1 * p1st
pS_F1 = pS_F * pF_1 * p1st

pS_M2 = pS_M * pM_2 * p2nd
pS_F2 = pS_F * pF_2 * p2nd

pS_M3 = pS_M * pM_3 * p3rd
pS_F3 = pS_F * pF_3 * p3rd

# create grouped bar chart to show the different probabilkities

# probabilities of survival | given a particular Gender and Ticket Class
mPcts = (pS_M1, pS_M2, pS_M3)
fPcts = (pS_F1, pS_F2, pS_F3)

barWidth = 0.35  # the width of the bars

# set the positions of the bars
p1 = np.arange(len(mPcts))
p2 = [x + barWidth for x in p1]

# Make the plot
plt.bar(p1, mPcts, color='blue', width=barWidth, edgecolor='white', label='Male')
plt.bar(p2, fPcts, color='pink', width=barWidth, edgecolor='white', label='Female')

# Add xticks on the middle of the group bars
plt.xlabel('Ticket Class', fontweight='bold')
plt.xticks([r + barWidth/2 for r in range(len(mPcts))], ['1st', '2nd', '3rd'])
 
# Create legend & Show graphic
plt.legend()
plt.title('Probabilities of Titanic Survival given Gender and Class')
